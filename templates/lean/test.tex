theorem mul_cancel_left_or {a b c : ℤ} (H : a * b = a * c) : a = 0 ∨ b = c :=
  have H2 : a * (b - c) = 0, by simp, 
  have H3 : a = 0 ∨ b - c = 0, from mul_eq_zero H2,
  or.imp_or_right H3 (assume H4 : b - c = 0, sub_eq_zero H4)
  \end{minted}
